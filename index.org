        for (i in pub) {

            // Bad IDs to skip
            if   (pub[i].arxivId_s == "physics/0610132"

               || pub[i].uri_s     == "https://hal.science/hal-03065465"
               || pub[i].uri_s     == "https://hal.science/hal-03065446"

                arXivId  = pub[i].arxivId_s;

                url = ""
            };
